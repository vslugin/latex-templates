% Определение рамки №1 ГОСТ
\newcommand{\gostframeone}{%
    \begin{tikzpicture}[remember picture, overlay]

    % Внутренняя рамка с отступом
    \draw[line width=0.5mm] ($(current page.south west) + (1cm,1cm)$)
    rectangle ($(current page.north east) - (1cm,1cm)$);

    \end{tikzpicture}%
}

% Определение рамки №2 ГОСТ
\newcommand{\gostframetwo}{%
    \begin{tikzpicture}[remember picture, overlay]

    % Внутренняя рамка с отступом
    \draw[line width=0.5mm] ($(current page.south west) + (1cm,1cm)$)
    rectangle ($(current page.north east) - (1cm,1cm)$);

    % Поля для основной надписи
    \draw[line width=0.5mm]
    ($(current page.south west) + (1cm,5cm)$) -- ++(19cm,0);
    \node[anchor=south west] at ($(current page.south west) + (1.2cm,1.2cm)$)
        {\small Основная надпись 2};
    \end{tikzpicture}%
}

% Определение рамки №3 ГОСТ
\newcommand{\gostframethree}{%
    \begin{tikzpicture}[remember picture, overlay]

    % Внутренняя рамка с отступом
    \draw[line width=0.5mm] ($(current page.south west) + (1cm,1cm)$)
    rectangle ($(current page.north east) - (1cm,1cm)$);

    % Поля для основной надписи
    \draw[line width=0.5mm]
    ($(current page.south west) + (1cm,2cm)$) -- ++(19cm,0);
    \node[anchor=south west] at ($(current page.south west) + (1.2cm,1.2cm)$)
        {\small Основная надпись 3};
    \end{tikzpicture}%
}



% Добавление рамки на каждую страницу
\AtBeginShipout{%
    \AtBeginShipoutUpperLeft{%
        \ifnum\thepage=1 \gostframeone % Если номер страницы равен 1
        \else\ifnum\thepage=2 \gostframetwo % Если номер страницы равен 2
        \else\ifnum\thepage=3 % Если номер страницы равен 3
            % не рисуем рамку для 3 страницы
            \else
                 \gostframethree % Для всех остальных страниц -- рамка по умолчанию
            \fi\fi\fi
    }%
}
