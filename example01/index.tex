% Установка отступов

\usepackage{fontspec}

% Настройка стиля страницы
\pagestyle{fancy}          % Устанавливает стиль "fancy"

\fancyfoot[C]{\thepage}    % Номер страницы внизу по центру

% Настройка содержания
\renewcommand{\cftsecfont}{\bfseries}
\renewcommand{\cftsecpagefont}{\bfseries}

% Титульная страница
\newcommand{\customtitle}[4]{%
    \begin{titlepage}
    \centering
    {\Large #1}\\[5cm] % Название организации
    {\huge\bfseries #2}\\[5cm] % Название документа
    {\Large Автор: #3}\\[3cm] % Автор
    {\Large Дата: #4}\\[3cm] % Дата
    \vfill
    \end{titlepage}
}
